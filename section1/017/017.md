## 1차 풀이

```javascript
function solution(s) {
  const flag = 1;
  for (let i = 0; i < s.length - 1; i++) {
    const base = s[i];

    if (base === flag) continue;
    for (let j = i + 1; j < s.length; j++) {
      const target = s[j];

      if (target === flag) continue;
      if (target === base) s[j] = flag;
    }
  }

  return s.filter((str) => str !== flag);
}
```

## 2차 풀이

```javascript
function solution(s) {
  const res = [];
  s.forEach((el, idx, arr) => {
    if (arr.indexOf(el) !== idx) return;
    res.push(el);
  });
  return res;
}
```
