## 문제

뒤집은 소수
N개의 자연수가 입력되면 각 자연수를 뒤집은 후 그 뒤집은 수가 소수이면 그 소수를 출력하
는 프로그램을 작성하세요. 예를 들어 32를 뒤집으면 23이고, 23은 소수이다. 그러면 23을 출
력한다. 단 910를 뒤집으면 19로 숫자화 해야 한다. 첫 자리부터의 연속된 0은 무시한다.
▣ 입력설명
첫 줄에 자연수의 개수 N(3<=N<=100)이 주어지고, 그 다음 줄에 N개의 자연수가 주어진다.
각 자연수의 크기는 100,000를 넘지 않는다.
▣ 출력설명
첫 줄에 뒤집은 소수를 출력합니다. 출력순서는 입력된 순서대로 출력합니다.
▣ 입력예제 1
9
32 55 62 20 250 370 200 30 100
▣ 출력예제 1
23 2 73 2 3

## 1차 풀이

```javascript
function isPrime(num) {
  let res = [];
  num.forEach((n) => {
    const transformedArr = [];
    //숫자 뒤집기
    [...n.toString()].forEach((char) => {
      transformedArr.unshift(char);
    });
    const reversedNum = Number(transformedArr.join(""));

    //엣지 케이스
    if (reversedNum === 2 || reversedNum === 3) {
      res.push(reversedNum);
      return;
    }
    if (reversedNum === 1) return;

    //그냥 sqrt가 아니라 Math.sqrt()임!!
    for (let i = 2; i <= Math.sqrt(reversedNum); i++) {
      if (reversedNum % i === 0) return;
    }
    res.push(reversedNum);
    return;
  });
  return res;
}

let arr = [32, 55, 62, 20, 250, 370, 200, 30, 100];
console.log(isPrime(arr));
```

## 2차 풀이

각 역할을 하는 함수로 분리해서 작업 실행.
그리고 먼저 짝수는 미리 제거한뒤에 for문에서 홀수씩 커지게 해 반복하는 횟수를 줄임.

```javascript
function reverseNum(num) {
  return Number([...num.toString()].reverse().join(""));
}

function isPrime(num) {
  if (num === 1) return false;
  if (num === 2) return true;
  if (num === 3) return true;

  if (num % 2 === 0) return false;
  for (let i = 3; i <= Math.sqrt(num); i += 2) {
    const namuzi = num % i; //나머지

    if (namuzi === 0) return false; //즉시 종료
  }
  return true;
}

function solution(arr) {
  const res = [];

  arr.forEach((num) => {
    const reversed = reverseNum(num);
    if (isPrime(reversed)) {
      res.push(reversed);
    }
  });

  return res;
}

let arr = [32, 55, 62, 20, 250, 370, 200, 30, 100];
console.log(solution(arr));
```
