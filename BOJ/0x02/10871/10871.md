# X보다 작은 수

## 설계

- 단순하게 입력된 input을 1-pass로 순환하면 구할 수 있겠지만(O(n)), 이번에는 시간복잡도를 고려하여 이분 탐색으로 풀어보자.

[1,23,42,5,14] -> 예시 input

- 1. 오름차순 정렬
- 2. 이분 탐색
  - 1. left와 right 존재.

```javascript
const fs = require("fs");

const input = fs.readFileSync("/dev/stdin").toString().trim().split("\n");

const [N, X] = input[0]
  .trim()
  .split(" ")
  .map((n) => Number(n));
const arr = input[1]
  .trim()
  .split(" ")
  .map((n) => Number(n));

function sol(x, arr) {
  const newArr = arr.sort((a, b) => a - b); //오름차순 정렬
  let left = 0;
  let right = arr.length - 1;
  let midIdx;
  while (left !== right) {
    midIdx = left + Math.floor((right - left) / 2);
    if (newArr[midIdx] < x) {
      left = midIdx + 1;
    } else if (newArr[midIdx] > x) {
      right = midIdx - 1;
    } else {
      break;
    }
  }

  const res = newArr.slice(0, midIdx).join(" ");
  console.log(res);
  return res;
}

sol(X, arr);
```
